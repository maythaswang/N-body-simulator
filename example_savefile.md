# SAVE FILE SAMPLE

```py
# [SAMPLE SAVE FILE]
# THIS IS THE SAMPLE FORMAT OF WHAT THE SAVE FILE WILL LOOK LIKE 
# BASED ON LOADING INSTRUCTION WE CAN LOAD STEP 3 DIRECTLY OR USE SETTING FROM STEP 2
# PLEASE ENSURE THAT THE NUMBER OF PARTICLES IN TOTAL ADDS UP TO <NUM_PARTICLES>
# EACH SECTION MUST BE ENCLOSED INSIDE A HEADER 'BEGIN...' AND 'END...'
#
# 
# [STEP_1] SETUP_SECTION
###################################
BEGIN_SETUP
USE_GPU <0,1>
USE_DEFAULT <0,1> 
DEFAULT_TEST_NUMBER <0-33>
GRAVITATIONAL_CONSTANT <FLOAT>
TIMESTEP_SIZE <FLOAT> 
INTEGRATOR <0,1>
NUM_PARTICLES <INT>
END_SETUP

# [STEP_2] PARTICLE_SETUP_SECTION
###################################
BEGIN_PARTICLE_CLUSTER_SETUP
PARTICLE_ADD <0: DISC> <N_PARTICLE> <OFFSET_X> <OFFSET_Y> <OFFSET_Z> <RADIUS> <WIDTH> <MIN_MASS> <MAX_MASS> <MIN_VELOCITY> <MAX_VELOCITY> <IS_SPIRAL> <IS_DENSE_CENTER>
PARTICLE_ADD <1: RANDOM> <N_PARTICLE> <OFFSET_X> <OFFSET_Y> <OFFSET_Z> <RADIUS> <MIN_MASS> <MAX_MASS> <MIN_VELOCITY> <MAX_VELOCITY>
PARTICLE_ADD <2: GLOBULAR_CLUSTER> <N_PARTICLE> <OFFSET_X> <OFFSET_Y> <OFFSET_Z> <RADIUS> <CENTER_RADIUS> <MIN_MASS> <MAX_MASS> <MIN_VELOCITY> <MAX_VELOCITY> <IS_SPIRAL> 
# OUTER ONLY IS DISABLED FOR GLOBULAR CLUSTER SO WE DON'T STORE IT
PARTICLE_ADD <3: SPHERE_SURFACE> <N_PARTICLE> <OFFSET_X> <OFFSET_Y> <OFFSET_Z> <RADIUS> <MIN_MASS> <MAX_MASS> <MIN_VELOCITY> <MAX_VELOCITY> <IS_SPIRAL>
END_PARTICLE_CLUSTER_SETUP


# [STEP_3] SAVED_SIMULATION
###################################
BEGIN_PARTICLE_DATA
# <EACH PARTICLE DATA DELIMITED BY SPACE> 
POSITION 
<(X,Y,Z)>...
VELOCITY
<(X,Y,Z)> <(X,Y,Z)>...
PREV_ACCELERATION
<(X,Y,Z)>...
MASS
m1 m2 m3.... 
END_PARTICLE_DATA
```
